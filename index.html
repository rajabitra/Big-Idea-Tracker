One-time SQL Setup
Run this once in Supabase → SQL Editor → New Query
-- Run this in Supabase SQL Editor (one time only)

create table if not exists ideas (
  id          bigserial primary key,
  title       text not null,
  description text,
  category    text,
  status      text default 'backlog',
  impact      text default 'medium',
  effort      text default 'medium',
  tags        text[],
  assignees   text[],
  milestones  jsonb default '[]',
  progress    integer default 0,
  notes       text,
  created_at  timestamptz default now(),
  updated_at  timestamptz default now()
);

-- Allow public read/write (for team sharing without login)
-- If you want login-protected access, set up Supabase Auth instead
alter table ideas enable row level security;

create policy "Public read"   on ideas for select using (true);
create policy "Public insert" on ideas for insert with check (true);
create policy "Public update" on ideas for update using (true);
create policy "Public delete" on ideas for delete using (true);
